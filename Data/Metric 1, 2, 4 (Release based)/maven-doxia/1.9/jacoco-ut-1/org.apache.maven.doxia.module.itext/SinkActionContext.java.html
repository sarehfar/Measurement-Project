    /**
     * <p>Getter for the field <code>currentAction</code>.</p>
     *
     * @return a int.
     */
    public int getCurrentAction()
    {
        //return currentAction;
        if ( stack.empty() )
        {
            return UNDEFINED;
        }
        else
        {
            return stack.peek();
        }
    }

    /**
     * release.
     */
    public void release()
    {
        //currentAction = -1;
        stack.pop();
    }

    /**
     * setAction.
     *
     * @param action a int.
     */
    public void setAction( int action )
    {
        //currentAction = action;

        stack.push( action );
    }
