import org.apache.maven.doxia.sink.Sink;

/**
 * SectionBlock
 *
 * @author <a href="mailto:jason@maven.org">Jason van Zyl</a>
 */
class SectionBlock
    implements Block
{
    private Block text;

    private int level;

    SectionBlock( Block text, int level )
    {
        this.text = text;
        this.level = level;
    }

    /** {@inheritDoc} */
    public void traverse( Sink sink )
    {
        if ( level == Sink.SECTION_LEVEL_1 )
        {
            sink.section1();
            sink.sectionTitle1();
        }
        else if ( level == Sink.SECTION_LEVEL_2 )
        {
            sink.section2();
            sink.sectionTitle2();
        }
        else if ( level == Sink.SECTION_LEVEL_3 )
        {
            sink.section3();
            sink.sectionTitle3();
        }
        else if ( level == Sink.SECTION_LEVEL_4 )
        {
            sink.section4();
            sink.sectionTitle4();
        }
        else if ( level == Sink.SECTION_LEVEL_5 )
        {
            sink.section5();
            sink.sectionTitle5();
        }

        this.text.traverse( sink );

        if ( level == Sink.SECTION_LEVEL_1 )
        {
            sink.sectionTitle1_();
            sink.section1_();
        }
        else if ( level == Sink.SECTION_LEVEL_2 )
        {
            sink.sectionTitle2_();
            sink.section2_();
        }
        else if ( level == Sink.SECTION_LEVEL_3 )
        {
            sink.sectionTitle3_();
            sink.section3_();

        }
        else if ( level == Sink.SECTION_LEVEL_4 )
        {
            sink.sectionTitle4_();
            sink.section4_();
        }
        else if ( level == Sink.SECTION_LEVEL_5 )
        {
            sink.sectionTitle5_();
            sink.section5_();
        }
    }
}
